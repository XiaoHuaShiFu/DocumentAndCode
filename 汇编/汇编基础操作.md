# 2、常用汇编代码段

1. 10进制输出

   ```assembly
   ;打印
   printf proc
   	push bp
   	mov bp, sp
   	push ax
   	push bx
   	push cx
   	push dx
   	
   	mov bx, [bp - 2]
   	cmp bx, 0
   	jge getDex
   	;打印"-"号
   	mov ah, 02h
   	mov dl, 2dh
   	int 21h
   	neg bx
   	
   getDex:
   	;循环得到每一位10进制
   	mov ax, bx
   	mov bx, 10
   	xor cx, cx
   	nextDex:
   		xor dx, dx
   		div bx
   		push dx
   		inc cx
   		;如果ax为0就退出循环
   		cmp ax, 0
   	jnz nextDex
   	
   	
   	;循环输出每一位10进制
   	mov ah, 02h
   	pushNextDex:
   		pop dx
   		add dx, 30h
   		int 21h
   	loop pushNextDex
   	
   	
   	pop dx
   	pop cx
   	pop bx
   	pop ax
   	pop bp
   	ret 2
   printf endp
   ```

   ```assembly
   ;打印并换行
   println proc

   	call printf

   	mov ah, 02h

   	mov dl, 0ah
   	int 21h

   	pop bp
   	ret 2
   println endp
   ```

3. 16进制输出

   - 实现原理

     ```xml
     循环左移4位
     取后4位
     判断并输出
     ```

   - 实现

     ```assembly
     ;打印，以16进制显示
     printf16 proc
     	push bp
     	mov bp, sp
     	push ax
     	push bx
     	push cx
     	push dx
     	
     	mov bx, [bp - 2]
     	
     getHex:
     	;循环得到每一位16进制
     	mov cx, 4
     	mov ah, 02h
     	nextHex:
     		push cx
     		mov cx, 4
     		rol bx, cl
     		pop cx
     		
     		mov dx, bx
     		and dx, 0fh
     		cmp dx, 10
     		;如果小于10
     		jl	l10
     		;否则（大于等于10）
     		jmp	ge10
     			
     		;小于10
     		l10:
     			add dx, 30h
     			int 21h	
     			jmp lend
     		;大于等于10
     		ge10:
     			add dx, 37h
     			int 21h	
     		;结束点
     		lend:
     	loop nextHex
     	
     	pop dx
     	pop cx
     	pop bx
     	pop ax
     	pop bp
     	ret 2
     printf16 endp
     ```

4. 打印数组（10进制）

   ```assembly
   ;打印数组
   printArray proc
   	push bp
   	mov bp, sp
   	push ax
   	push bx
   	push cx
   	push dx
   	push si
   		
   	
   	;左"["
   	mov dl, '['
   	call putchar
   	
   	mov cx, [bp + 4]
   	;如果数组没有元素，直接跳到最后
   	cmp cx, 0
   	jz	end0
   	sub cx, 1
   	mov si, [bp + 6]
   	
   	
   	;第一个元素
   	mov ax, [si]
   	push ax
   	call printInt
   	add si, 2
   	;打印剩下的元素
   	printElements:
   		mov dl, ','
   		call putchar
   		mov dl, ' '
   		call putchar
   		
   		mov ax, [si]
   		push ax
   		call printInt
   		add si, 2
   	loop printElements
   	
   	;最后
   	end0:
   	;右"]"
   	mov dl, ']'
   	call putchar
   	
   	pop si
   	pop dx
   	pop cx
   	pop bx
   	pop ax
   	pop bp
   	ret 4
   printArray endp
   
   ;推出一个字符
   putchar proc
   	push ax
   	mov ah, 02h
   	int 21h
   	pop ax
   	ret
   putchar endp
   
   ;打印
   printInt proc
   	push bp
   	mov bp, sp
   	push ax
   	push bx
   	push cx
   	push dx
   	
   	mov bx, [bp - 2]
   	cmp bx, 0
   	jge getDex
   	;打印"-"号
   	mov ah, 02h
   	mov dl, 2dh
   	int 21h
   	neg bx
   	
   getDex:
   	;循环得到每一位10进制
   	mov ax, bx
   	mov bx, 10
   	xor cx, cx
   	nextDex:
   		xor dx, dx
   		div bx
   		push dx
   		inc cx
   		;如果ax为0就退出循环
   		cmp ax, 0
   	jnz nextDex
   	
   	
   	;循环输出每一位10进制
   	pushNextDex:
   		pop dx
   		add dx, 30h
   		call putchar
   	loop pushNextDex
   	
   	
   	pop dx
   	pop cx
   	pop bx
   	pop ax
   	pop bp
   	ret 2
   printInt endp
   ```
